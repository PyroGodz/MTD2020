uwhole func myuf(uwhole a)
[
uwhole t : 0033.
t : t + a.
repay t. 
]

master
[
logic check : true.
meout(check).
string  str : "hello".
string sss : strdup(str).
meout(sss).
uwhole m : myuf(3).
meout(m).
uwhole inverse : ~ 15. 
meout(inverse).
uwhole or : 55 | 80.
uwhole and : 56 & 76.
if (or > and)
[
meout(or).
]
else
[
meout(and).
]
string n : "привет".
string s : "котята".
string ns : stradd(n, s).
meout(ns).
uwhole j : ((44 + 4) / 7).
meout(j).
]
